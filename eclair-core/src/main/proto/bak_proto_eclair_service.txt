syntax = "proto3";

package fr.acinq.eclair;

service RpcService {
  // getinfo API
  rpc GetInfo (RpcGetInfoRequest) returns (RpcGetInfoResponse) { }
}

// The request message for getting node info.
message RpcGetInfoRequest { }

// The response message containing the info
message RpcGetInfoResponse {
  string nodeId = 1;
  string alias = 2;
  int32 port = 3;
  string chainHash = 4;
  int32 blockHeight = 5;
}
